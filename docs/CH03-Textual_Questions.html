Chapter 3 - Textual Questions

  1. Write a C program to find the sum and average of three given numbers.

    #include 

int main() {
    int num1, num2, num3, sum = 0;
    float average = 0;

    printf("Enter 3 numbers : ");
    scanf("%d %d %d", &num1, &num2, &num3);

    sum = num1 + num2 + num3;
    average = (num1 + num2 + num3) / 3.0;

    printf("Sum : %d \nAverage : %6.2f\n", sum, average);
    return 0;
}

  2. Write a C program to convert temperature in °C to °F using the relation °F = 1.8°C + 32.

    #include 

int main() {
    float cel, far;

    printf("Enter Celsius : ");
    scanf("%f", &cel);

    far = 1.8 * cel + 32;

    printf("Farenheit : %6.2f°F\n", far);
    return 0;
}

  3. Write a C program to find the area and circumference of a circle of radius r.

    #include 

int main() {
    float r, area, circumference;

    printf("Enter Radius of the Circle : ");
    scanf("%f", &r);

    area = 22.0/7.0 * r * r; 
    circumference = 2 * 22.0/7.0 * r;

    printf("Area of circle : %.2f\n", area);
    printf("Circumference of Circle : %.2f\n", circumference);
    return 0;
}
